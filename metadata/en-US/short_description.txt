        async function initiateBLEScan() {
            const logEl = document.getElementById('status-log');
            const listEl = document.getElementById('device-list');

            if (!navigator.bluetooth) {
                logEl.innerHTML = "<span style='color: var(--accent-red);'>Web Bluetooth API not supported on this browser/platform.</span>";
                return;
            }

            try {
                logEl.innerText = "Requesting Bluetooth hardware permission...";
                const device = await navigator.bluetooth.requestDevice({
                    acceptAllDevices: true,
                    optionalServices: ['generic_access', 'battery_service']
                });

                logEl.innerText = `Connected to: ${device.name || 'Unnamed Device'} [${device.id}]`;

                const nodeDiv = document.createElement('div');
                nodeDiv.className = 'node-item';
                nodeDiv.innerHTML = `
                    <span class="badge badge-standard">BLE PERIPHERAL</span>
                    <strong style="display:block; margin-top: 4px;">${device.name || 'Unknown Device'}</strong>
                    <small style="color: var(--text-dim);">Hardware ID: ${device.id}</small><br>
                    <small style="color: var(--accent-amber);">GATT State: Connected</small>
                `;
                listEl.appendChild(nodeDiv);

            } catch (error) {
                logEl.innerHTML = `<span style='color: var(--accent-amber);'>Scan status: ${error.message}</span>`;
            }
        }
